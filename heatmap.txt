  function csvToJson(csvText) {
    const rows = parseCSV(csvText);
    if (rows.length < 2) return [];

    const headers = rows[0].map(h => cleanHeader(h));

    return rows.slice(1)
      .filter(row => row.some(value => value && value.trim() !== ""))
      .map(row => {
        const obj = {};

        headers.forEach((header, index) => {
          obj[header] = row[index] ? row[index].trim() : "";
        });

        return normalizeRecord(obj);
      })
      .filter(x => x.assignedTo && x.start && x.finish);
  }

  function cleanHeader(value) {
    return value
      .replace(/^\uFEFF/, "")
      .trim();
  }

  function normalizeRecord(row) {
    return {
      id: row["ID"] || row["Id"] || row["Work Item Id"] || "",
      type: row["Work Item Type"] || row["Type"] || "",
      title: row["Title"] || "",
      assignedTo: cleanupAssignedTo(row["Assigned To"] || row["AssignedTo"] || ""),
      project: row["Project"] || row["Team Project"] || row["Area Path"] || "",
      areaPath: row["Area Path"] || "",
      iterationPath: row["Iteration Path"] || "",
      start: formatDate(row["Start Date"] || row["StartDate"] || row["Created Date"] || ""),
      finish: formatDate(row["Target Date"] || row["Finish Date"] || row["End Date"] || row["Closed Date"] || ""),
      state: row["State"] || "",
      parent: row["Parent"] || row["Parent ID"] || "",
      tags: row["Tags"] || ""
    };
  }

  function cleanupAssignedTo(value) {
    // Handles formats like "Kiran Joshi <kiran.joshi@domain.com>"
    return value.replace(/<[^>]*>/g, "").trim();
  }

  function formatDate(value) {
    if (!value) return "";

    const cleaned = value.trim();

    // Handles 2026-07-06
    if (/^\d{4}-\d{2}-\d{2}/.test(cleaned)) {
      return cleaned.substring(0, 10);
    }

    // Handles 06/07/2026 or 7/6/2026 depending on browser locale.
    const parsed = new Date(cleaned);
    if (!isNaN(parsed.getTime())) {
      return parsed.toISOString().substring(0, 10);
    }

    return "";
  }

  function parseCSV(text) {
    const rows = [];
    let row = [];
    let current = "";
    let insideQuotes = false;

    for (let i = 0; i < text.length; i++) {
      const char = text[i];
      const next = text[i + 1];

      if (char === '"' && insideQuotes && next === '"') {
        current += '"';
        i++;
      } else if (char === '"') {
        insideQuotes = !insideQuotes;
      } else if (char === "," && !insideQuotes) {
        row.push(current);
        current = "";
      } else if ((char === "\n" || char === "\r") && !insideQuotes) {
        if (current !== "" || row.length > 0) {
          row.push(current);
          rows.push(row);
          row = [];
          current = "";
        }

        if (char === "\r" && next === "\n") i++;
      } else {
        current += char;
      }
    }

    if (current !== "" || row.length > 0) {
      row.push(current);
      rows.push(row);
    }

    return rows;
  }

  function render(records) {
    const validRecords = records
      .filter(x => x.assignedTo && x.start && x.finish)
      .sort((a, b) => {
        if (a.assignedTo !== b.assignedTo) return a.assignedTo.localeCompare(b.assignedTo);
        return new Date(a.start) - new Date(b.start);
      });

    document.getElementById("empty").style.display = validRecords.length ? "none" : "block";

    if (!validRecords.length) {
      document.getElementById("summary").innerHTML = "No valid records found. Please check mandatory columns and date values.";
      document.getElementById("gantt").innerHTML = "";
      document.getElementById("legend").innerHTML = "";
      return;
    }

    const weeks = buildWeeks(validRecords);
    const grouped = groupByResource(validRecords);
    const resourceColors = assignColors(Object.keys(grouped));

    renderSummary(validRecords, grouped, weeks);
    renderLegend(resourceColors);
    renderTable(grouped, weeks, resourceColors);
  }

  function buildWeeks(records) {
    const minDate = new Date(Math.min(...records.map(x => new Date(x.start))));
    const maxDate = new Date(Math.max(...records.map(x => new Date(x.finish))));

    const firstMonday = getMonday(minDate);
    const weeks = [];

    let current = new Date(firstMonday);
    let weekNo = 1;

    while (current <= maxDate) {
      const start = new Date(current);
      const end = new Date(current);
      end.setDate(end.getDate() + 6);

      weeks.push({
        index: weekNo,
        label: "W" + weekNo + "<br>" + formatDisplayDate(start),
        start,
        end
      });

      current.setDate(current.getDate() + 7);
      weekNo++;
    }

    return weeks;
  }

  function getMonday(date) {
    const d = new Date(date);
    const day = d.getDay();
    const diff = d.getDate() - day + (day === 0 ? -6 : 1);
    d.setDate(diff);
    d.setHours(0, 0, 0, 0);
    return d;
  }

  function formatDisplayDate(date) {
    return date.toLocaleDateString("en-GB", { day: "2-digit", month: "short" });
  }

  function groupByResource(records) {
    return records.reduce((acc, item) => {
      if (!acc[item.assignedTo]) acc[item.assignedTo] = [];
      acc[item.assignedTo].push(item);
      return acc;
    }, {});
  }

  function assignColors(resources) {
    const colors = {};

    resources.forEach((resource, index) => {
      colors[resource] = colorPalette[index % colorPalette.length];
    });

    return colors;
  }

  function renderSummary(records, grouped, weeks) {
    document.getElementById("summary").innerHTML =
      `<b>${records.length}</b> work item allocation(s), ` +
      `<b>${Object.keys(grouped).length}</b> resource(s), ` +
      `<b>${weeks.length}</b> week(s) displayed.`;
  }

  function renderLegend(resourceColors) {
    let html = "";

    Object.keys(resourceColors).forEach(resource => {
      html += `
        <div class="legend-item">
          <span class="legend-box" style="background:${resourceColors[resource]}"></span>
          ${escapeHtml(resource)}
        </div>`;
    });

    document.getElementById("legend").innerHTML = html;
  }

  function renderTable(grouped, weeks, resourceColors) {
    let html = "<thead><tr>";
    html += "<th class='resource-header resource-col'>Resource</th>";
    html += "<th class='item-header item-col'>ADO Item / Project</th>";

    weeks.forEach(week => {
      html += `<th>${week.label}</th>`;
    });

    html += "</tr></thead><tbody>";

    Object.keys(grouped).forEach(resource => {
      const items = grouped[resource];
      const color = resourceColors[resource];

      items.forEach((row, rowIndex) => {
        html += "<tr>";

        if (rowIndex === 0) {
          html += `
            <td class="resource-col"
                rowspan="${items.length}"
                style="background:${color}">
              ${escapeHtml(resource)}<br>
              <span style="font-size:11px;font-weight:normal;">${items.length} item(s)</span>
            </td>`;
        }

        html += `
          <td class="item-col">
            ${escapeHtml(row.type)} ${escapeHtml(row.id)} - ${escapeHtml(row.title)}
            <span class="project">${escapeHtml(row.project)} | ${escapeHtml(row.state)}</span>
          </td>`;

        weeks.forEach((week, weekIndex) => {
          html += "<td>";

          if (isBarStart(row, week)) {
            const spanWeeks = calculateWeekSpan(row, weeks, weekIndex);
            const width = (spanWeeks * weekWidth) - 14;

            html += `
              <div class="bar"
                   style="width:${width}px;background:${color}"
                   title="${escapeHtml(resource)} | ${escapeHtml(row.project)} | ${escapeHtml(row.title)} | ${row.start} to ${row.finish}">
                ${escapeHtml(row.title)}
              </div>`;
          }

          html += "</td>";
        });

        html += "</tr>";
      });
    });

    html += "</tbody>";

    document.getElementById("gantt").innerHTML = html;
  }

  function isBarStart(row, week) {
    const start = new Date(row.start);
    return start >= week.start && start <= week.end;
  }

  function calculateWeekSpan(row, weeks, startWeekIndex) {
    const finish = new Date(row.finish);
    let span = 1;

    for (let i = startWeekIndex + 1; i < weeks.length; i++) {
      if (finish >= weeks[i].start) {
        span++;
      }
    }

    return span;
  }

  function escapeHtml(value) {
    return String(value || "")
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#039;");
  }
